Q1. Online Exam Portal (Basic)

class User {
    String username;
    String password;

    User(String username, String password) {
        this.username = username;
        this.password = password;
    }

    void login() {
        System.out.println("Generic user login...");
    }
}

class Student extends User {
    boolean enrolled;

    Student(String username, String password, boolean enrolled) {
        super(username, password);
        this.enrolled = enrolled;
    }

    @Override
    void login() {
        if (enrolled) {
            System.out.println("Student " + username + " logged in. Exams available.");
        } else {
            System.out.println("Access denied: Student not enrolled.");
        }
    }
}

class Admin extends User {
    boolean isAdmin;

    Admin(String username, String password, boolean isAdmin) {
        super(username, password);
        this.isAdmin = isAdmin;
    }

    @Override
    void login() {
        if (isAdmin) {
            System.out.println("Admin " + username + " logged in. Manage exams.");
        } else {
            System.out.println("Access denied: Not an admin.");
        }
    }
}

public class OnlineExamPortal {
    public static void main(String[] args) {
        Student s = new Student("john123", "pass", true);
        s.login();

        Admin a = new Admin("admin1", "adminpass", true);
        a.login();
    }
}
